functions

function R = mrz(theta)
    R = [cos(theta), sin(theta), 0;
        -sin(theta), cos(theta), 0;
        0, 0, 1];
end
function R = mrx(theta)
    R = [1,0,0;
        0,cos(theta), sin(theta);
        0,-sin(theta),cos(theta)];
end
function R = mry(theta)
    R = [cos(theta),0,-sin(theta);
        0,1, 0;
        sin(theta),0,cos(theta)];
end
